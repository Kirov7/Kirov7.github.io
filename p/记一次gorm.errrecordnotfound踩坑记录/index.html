	out := &ServiceInfo{}
	resultFind := tx.WithContext(c).Where(search).Find(out)
	if resultFind.RowsAffected < 1 {
		err := resultFind.AddError(gorm.ErrRecordNotFound)
		if err != nil {
			return nil, err
		}
	}
	err := resultFind.Error
	if err != nil {
		return nil, err
	}
	return out, nil
}
